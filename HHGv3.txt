Models ------ dephasing L_i = n_i (local density dephasing, original model). Conserves total particle number; destroys phase coherence. momentum_relaxing L_{ij,σ} = c†_i c_j for all bonds (i,j) and spin σ. Models momentum relaxation without particle-number change. More physical for electron-impurity scattering. electron_phonon L_{i,σ} = c_{i,σ} + c†_{i,σ} (amplitude damping / phonon coupling). """ model = self.config.dissipation_model L = [] if model == "dephasing": # L_i = sqrt(γ) n_i for n_i in n_ops: L.append(np.sqrt(self.config.gamma) * n_i) elif model == "momentum_relaxing": # L_{ij,σ} = sqrt(γ) c†_i c_j for every directed bond, both spins # Uses only bonds in the graph (not all pairs) for locality. for i, j in self.bonds: for c_list in (c_up, c_dn): L.append(np.sqrt(self.config.gamma) * c_list[i].dag() * c_list[j]) L.append(np.sqrt(self.config.gamma) * c_list[j].dag() * c_list[i]) elif model == "electron_phonon": # L_{i,σ} = sqrt(γ) (c_{i,σ} + c†_{i,σ}) — amplitude bath for i in range(self.N): for c_list in (c_up, c_dn): L.append( np.sqrt(self.config.gamma) * (c_list[i] + c_list[i].dag()) ) return L # ------------------------------------------------------------------ # Precomputed bond data # ------------------------------------------------------------------ def _precompute_bond_hop_data(self) -> List[dict]: data = [] for i, j in self.bonds: d_vec = self.pos[j] - self.pos[i] for spin_label, c_list in (("up", self.ops["c_up"]), ("dn", self.ops["c_dn"])): op_plus = c_list[i].dag() * c_list[j] op_minus = c_list[j].dag() * c_list[i] data.append({ "i": i, "j": j, "spin": spin_label, "d_vec": d_vec.copy(), "op_plus": op_plus, "op_minus": op_minus, }) return data # ------------------------------------------------------------------ # Pulse # ------------------------------------------------------------------ def _pulse_duration(self) -> float: return 2 * np.pi * self.config.pulse_cycles / self.config.omega def _envelope(self, tau: float) -> float: T = self._pulse_duration() return np.sin(np.pi * tau / T) ** 2 if 0.0 < tau < T else 0.0 def _vector_potential(self, tau: float) -> np.ndarray: """A(t) = (E0/ω) sin(ωt) f(t) p̂ (velocity gauge).""" If half_filling is True: project onto the half-filled sector before returning the density matrix. Diamagnetic term: J_dia = −A(t) · ⟨N⟩ # A is a 2-vector; ⟨N⟩ is scalar → element-wise product gives 2-vector n_total = self._compute_total_density(rho) J_dia = -A * n_total # shape (2,) J += J_dia return np.real(J) # ------------------------------------------------------------------ # Extra observables # ------------------------------------------------------------------ def _compute_observables(self, rho: qt.Qobj) -> Dict: site_density = np.array([qt.expect(n_i, rho) for n_i in self.ops["n_total"]]) double_occ = float(sum(qt.expect(d_i, rho) for d_i in self.ops["double_occ"])) return { "site_density": site_density, "total_density": float(site_density.sum()), "double_occ": double_occ, } # ------------------------------------------------------------------ # Spectrum # ------------------------------------------------------------------ def _compute_hhg_spectrum( self, tlist: np.ndarray, J_t: np.ndarray, ) -> Tuple[np.ndarray, np.ndarray]: """HHG spectrum |ω² J(ω)|² with Hann window + zero-padding.""" window = np.hanning(len(J_t)) J_win = J_t * window n_pad = len(J_t) * self.config.zero_pad_factor dt = tlist[1] - tlist[0] J_fft = np.fft.fftshift(np.fft.fft(J_win, n=n_pad)) freq = np.fft.fftshift(np.fft.fftfreq(n_pad, d=dt)) mask = freq > 1e-8 freq_pos = freq[mask] spectrum = np.abs(freq_pos ** 2 * J_fft[mask]) ** 2 return freq_pos, spectrum # ------------------------------------------------------------------ # Main run method # ------------------------------------------------------------------ def run( self, use_mcsolve: bool = False, ntraj: int = 200, track_observables: bool = True, ) -> Dict: """Propagate the density matrix and collect observables.""" rho0 = self._get_ground_state() c_ops = [qt.sqrt(self.config.gamma) * L for L in self.ops["L_list"]] print(" Running solver …") if use_mcsolve: result = qt.mcsolve( self._H_t, rho0, self.tlist, c_ops, ntraj=ntraj, options=qt.Options(nsteps=1_000_000), ) states = [ sum(qt.ket2dm(result.states[tr][i]) for tr in range(ntraj)) / ntraj for i in range(len(self.tlist)) ] else: result = qt.mesolve( self._H_t, rho0, self.tlist, c_ops, options=qt.Options(nsteps=1_000_000), ) states = result.states Jx = np.zeros(len(self.tlist)) Jy = np.zeros(len(self.tlist)) double_occ_t = np.zeros(len(self.tlist)) total_density_t = np.zeros(len(self.tlist)) site_density_t = np.zeros((len(self.tlist), self.N)) for i, (t, rho) in enumerate(zip(self.tlist, states)): J = self._compute_current(t, rho) Jx[i] = J[0] Jy[i] = J[1] if track_observables: obs = self._compute_observables(rho) double_occ_t[i] = obs["double_occ"] total_density_t[i] = obs["total_density"] site_density_t[i] = obs["site_density"] freq_x, spectrum_x = self._compute_hhg_spectrum(self.tlist, Jx) freq_y, spectrum_y = self._compute_hhg_spectrum(self.tlist, Jy) output = { "tlist": self.tlist, "Jx": Jx, "Jy": Jy, "freq_x": freq_x, "spectrum_x": spectrum_x, "freq_y": freq_y, "spectrum_y": spectrum_y, "double_occ_t": double_occ_t, "total_density_t": total_density_t, "site_density_t": site_density_t, "config": self.config, "N": self.N, "lattice_type": self.config.lattice_type, "periodic": self.config.periodic, } if self.config.save_hdf5: self._save_hdf5(output) return output # ------------------------------------------------------------------ # HDF5 output # ------------------------------------------------------------------ def _save_hdf5(self, result: Dict) -> None: path = self.config.hdf5_path with h5py.File(path, "w") as f: for key in ( "tlist", "Jx", "Jy", "freq_x", "spectrum_x", "freq_y", "spectrum_y", "double_occ_t", "total_density_t", "site_density_t", ): f.create_dataset(key, data=result[key]) grp = f.create_group("config") for k, v in vars(self.config).items(): try: grp.attrs[k] = v except TypeError: grp.attrs[k] = str(v) print(f" Results saved → {path}") # ------------------------------------------------------------------ # Visualisation # ------------------------------------------------------------------ def plot_lattice(self, ax: Optional[plt.Axes] = None) -> plt.Axes: if ax is None: _, ax = plt.subplots(figsize=(5, 5)) pos_dict = {i: self.pos[i, :2] for i in range(self.N)} nx.draw( self.graph, pos=pos_dict, ax=ax, with_labels=True, node_color="steelblue", node_size=400, font_color="white", edge_color="gray", ) ax.set_title( f"{self.config.lattice_type} (N={self.N}, " f"PBC={self.config.periodic})" ) return ax # ============================================================================= # Convenience plotting # ============================================================================= def plot_results( results: List[Dict], titles: Optional[List[str]] = None, ) -> plt.Figure: n = len(results) titles = titles or [r.get("lattice_type", f"Run {i}") for i, r in enumerate(results)] fig, axs = plt.subplots(n, 3, figsize=(15, 4 * n)) if n == 1: axs = axs[np.newaxis, :] for idx, (res, title) in enumerate(zip(results, titles)): axs[idx, 0].plot(res["tlist"], res["Jx"], label="Jx") axs[idx, 0].plot(res["tlist"], res["Jy"], label="Jy", ls="--") axs[idx, 0].set(title=f"{title} – J(t)", xlabel="t", ylabel="J") axs[idx, 0].legend(); axs[idx, 0].grid(True) axs[idx, 1].semilogy(res["freq_x"], res["spectrum_x"]) axs[idx, 1].set(title=f"{title} – Spectrum Jx", xlabel="ω/ω₀", ylabel="|ω²Jx(ω)|²") axs[idx, 1].grid(True) if np.any(res["double_occ_t"] != 0): axs[idx, 2].plot(res["tlist"], res["double_occ_t"], color="crimson") axs[idx, 2].set(title=f"{title} – Double occ.", xlabel="t", ylabel="D(t)") axs[idx, 2].grid(True) else: axs[idx, 2].axis("off") plt.tight_layout() return fig # Changelog (v2 → v3)

Improvement 1 — Imaginary-time evolution: Euler → RK4

Problem: The v2 Euler stepper ψ ← ψ − dτ H ψ is first-order accurate (O(dτ)) and can diverge when dτ · ‖H‖ > 1. For large U/t, the spectral radius of H is large and either the step size must be tiny (slow) or the evolution blows up.

Fix: _get_ground_state() now uses a 4th-order Runge-Kutta (RK4) imaginary-time stepper:

k1 = −H ψ k2 = −H (ψ + dτ/2 · k1) k3 = −H (ψ + dτ/2 · k2) k4 = −H (ψ + dτ · k3) ψ_new = ψ + (dτ/6)(k1 + 2k2 + 2k3 + k4) ψ_new /= ‖ψ_new‖ 

Error per step is O(dτ⁵); global error O(dτ⁴). A stability warning is emitted if dτ · spectral_radius > 1.5.

For N ≤ 4, exact diagonalisation (ED) is always used — there is no benefit to imaginary-time integration for very small systems.

Improvement 2 — Half-filling enforcement

Problem: The Hubbard model's most interesting physics (Mott insulator transition, spin-charge separation) occurs at half-filling (⟨N⟩ = N_sites, N_up = N_dn = N/2). Without sector projection the ground state can have any filling depending on U and the lattice.

Fix: New SimConfig.half_filling: bool = False flag. When enabled, _project_to_half_filling():

Diagonalises H(t=0).

Keeps eigenstates where |⟨N_up⟩ − N/2| < 0.05 and |⟨N_dn⟩ − N/2| < 0.05.

Returns the lowest-energy eigenstate in that sector.

A warning is issued if no half-filled states are found.

Note: the electron_phonon dissipation model does not conserve N — using half_filling=True with it will enforce half-filling in the ground state but the density will drift during time evolution. This is physically meaningful for polaron physics but should be used with awareness.

Improvement 3 — Richer dissipation models

Problem: Local dephasing L_i = √γ n_i destroys phase coherence but does not relax momentum. Real materials have electron-impurity scattering (momentum relaxation) and electron-phonon coupling.

Fix: New SimConfig.dissipation_model string with three options:

ModelCollapse operatorsPhysical picture"dephasing" (default)L_i = √γ n_iLocal density dephasing; preserves N"momentum_relaxing"L_{ij,σ} = √γ c†_i c_j (all bonds, both directions, both spins)Inter-site hopping bath; momentum relaxation without N change"electron_phonon"L_{i,σ} = √γ (c_{iσ} + c†_{iσ})Amplitude damping; allows N fluctuation (polaron / strong e-ph) 

Improvement 4 — Performance guidance for large N

Problem: For N > 6–8, the full density-matrix dimension 4^N makes mesolve prohibitively slow.

Fixes:

use_sparse: bool = True flag in SimConfig; operators built with QuTiP's sparse backend (default).

Warning messages now include concrete upgrade paths: 

quimb with MPO/DMRG backend

ITensors.jl (Julia) via subprocess

qutip-qip tensor-network branch (experimental)

Tiered warnings: N > 6 → suggestion; N > 8 → strong warning.

Improvement 5 — Corrected PBC for triangular and kagome

Problem: v2 PBC in _lattice_triangular wrapped along Cartesian rows/columns, which is incorrect for a triangular torus (the two primitive lattice vectors are not orthogonal). Similarly, the 1D kagome strip had no 2D generalisation.

Fixes:

Triangular: Primitive vectors a1 = a[1,0], a2 = a[½,√3/2]. Sites are indexed as (r, c) in the a1–a2 frame. PBC wraps r % rows, c % cols. The three nearest-neighbour vectors are (0,+1), (+1,0), (+1,−1) in lattice coordinates — exactly the triangular lattice NN bonds.

Kagome: The unit cell (A, B, C) is tiled on an n_rows × n_cols supercell in the a1–a2 frame. Inter-cell bonds connect B → A_right and C → A_up with cell_idx() wrapping via modulo — producing a correct kagome torus for any n_rows × n_cols × 3 site count.

Improvement 6 — Units documentation

_units_summary() classmethod prints a formatted table of all units.

Extended SimConfig docstring includes current and HHG-intensity units.

_compute_current() docstring now explains the physical origin of J_dia and its role in the f-sum rule.

Comment in _compute_current clarified: J_dia = -A * n_total where A is a 2-vector and n_total is a scalar, giving a 2-vector result via NumPy broadcasting.

Known Limitations / Future Work

Scalability: For N > 8–10, the QuTiP full density-matrix approach is intractable. The recommended upgrade is quimb (MPO + DMRG/TDVP) or ITensors.jl.

Dissipation: The three bath models are phenomenological. A more physical Ohmic electron-phonon bath would require a frequency-dependent spectral density and Born-Markov treatment (e.g., Redfield or HEOM).

PBC geometry: Triangular and kagome PBC are now correct for rectangular supercells in the a1–a2 frame. Truly periodic tiling of an arbitrary cluster (e.g., for momentum-resolved observables) requires additional supercell construction.

Current for electron_phonon bath: The diamagnetic term assumes a fixed ⟨N⟩; for the amplitude-damping bath, ⟨N⟩ drifts in time and this is correctly handled since _compute_total_density is called at each time step.
